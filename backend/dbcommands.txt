use BusNirikshan

db.createCollection("users", {
  validator: {
    $jsonSchema: {
      bsonType: "object",
      required: ["name", "email", "passwordHash", "role", "createdAt", "updatedAt"],
      properties: {
        name: {
          bsonType: "string",
          description: "must be a string and is required"
        },
        email: {
          bsonType: "string",
          pattern: "^.+@.+\\..+$",
          description: "must be a valid email string and is required — stored as lowercase"
        },
        passwordHash: {
          bsonType: "string",
          description: "must be a string and is required"
        },
        role: {
          enum: ["passenger", "driver", "admin"],
          description: "can only be 'passenger', 'driver', or 'admin' and is required"
        },
        rtc: {
          bsonType: ["string", "null"],
          enum: ["GSRTC", "MSRTC", "RSRTC", null],
          description: "can only be one of the specified RTCs or null"
        },
        isActive: {
          bsonType: "bool",
          description: "must be a boolean"
        },
        createdAt: {
          bsonType: "date",
          description: "must be a date and is required"
        },
        updatedAt: {
          bsonType: "date",
          description: "must be a date and is required"
        }
      }
    }
  }
})

// 1. Create unique index on email
db.users.createIndex({ email: 1 }, { unique: true })

// 2. Create standard index on role
db.users.createIndex({ role: 1 })





db.createCollection("refreshtokens", {
  validator: {
    $jsonSchema: {
      bsonType: "object",
      required: ["userId", "email", "token", "expiresAt", "createdAt"],
      properties: {
        userId: {
          bsonType: "objectId",
          description: "must be an ObjectId referencing the user and is required"
        },
        email: {
          bsonType: "string",
          pattern: "^.+@.+\\..+$", // Basic email regex validation
          description: "must be a valid email string and is required"
        },
        token: {
          bsonType: "string",
          description: "must be a string (hashed token) and is required"
        },
        expiresAt: {
          bsonType: "date",
          description: "must be a date and is required for TTL"
        },
        createdAt: {
          bsonType: "date",
          description: "must be a date and is required"
        }
      }
    }
  }
})

// 1. Unique index on the token to prevent duplicates
db.refreshtokens.createIndex({ token: 1 }, { unique: true })

// 2. TTL Index: Tells MongoDB to delete the document when the current time reaches the 'expiresAt' value
db.refreshtokens.createIndex({ expiresAt: 1 }, { expireAfterSeconds: 0 })

// 3. Index on userId to quickly find all active sessions for a specific user (e.g. "log out all devices")
db.refreshtokens.createIndex({ userId: 1 })


db.createCollection("passwordresettokens", {
  validator: {
    $jsonSchema: {
      bsonType: "object",
      required: ["userId", "email", "token", "expiresAt", "used", "createdAt"],
      properties: {
        userId: {
          bsonType: "objectId",
          description: "must be an ObjectId referencing the user and is required"
        },
        email: {
          bsonType: "string",
          pattern: "^.+@.+\\..+$", // Basic email regex validation
          description: "must be a valid email string and is required"
        },

        token: {
          bsonType: "string",
          description: "must be a string (hashed token) and is required"
        },
        expiresAt: {
          bsonType: "date",
          description: "must be a date and is required for TTL"
        },
        used: {
          bsonType: "bool",
          description: "must be a boolean to track if the token was consumed"
        },
        createdAt: {
          bsonType: "date",
          description: "must be a date and is required"
        }
      }
    }
  }
})

// 1. Unique index to ensure tokens cannot be duplicated
db.passwordresettokens.createIndex({ token: 1 }, { unique: true })

// 2. TTL Index: Tells MongoDB to automatically delete the document when the current time reaches 'expiresAt'
db.passwordresettokens.createIndex({ expiresAt: 1 }, { expireAfterSeconds: 0 })

// 3. Index on userId to quickly find all tokens for a specific user
db.passwordresettokens.createIndex({ userId: 1 })

// 4. Compound index: powers deleteMany({ userId, used: false }) in auth.js — avoids full collection scan
db.passwordresettokens.createIndex({ userId: 1, used: 1 })


db.createCollection("stops", {
  validator: {
    $jsonSchema: {
      bsonType: "object",
      required: ["name", "city", "state", "rtc", "location", "createdAt", "updatedAt"],
      properties: {
        name: {
          bsonType: "string",
          description: "must be a string and is required"
        },
        city: {
          bsonType: "string",
          description: "must be a string and is required"
        },
        state: {
          bsonType: "string",
          description: "must be a string and is required"
        },
        rtc: {
          bsonType: "array",
          minItems: 1, // Ensure the array isn't empty
          description: "must be an array containing at least one valid RTC string",
          items: {
            bsonType: "string",
            enum: ["GSRTC", "MSRTC", "RSRTC", "ALL", "NONE"]
          }
        },
        location: {
          bsonType: "object",
          required: ["type", "coordinates"],
          properties: {
            type: {
              enum: ["Point"],
              description: "must be the string 'Point'"
            },
            coordinates: {
              bsonType: "array",
              minItems: 2,
              maxItems: 2,
              items: {
                bsonType: ["double", "int", "long", "decimal"]
              },
              description: "must be an array of exactly two numbers: [longitude, latitude]"
            }
          }
        },
        isActive: {
          bsonType: "bool",
          description: "must be a boolean"
        },
        createdAt: {
          bsonType: "date",
          description: "must be a date and is required"
        },
        updatedAt: {
          bsonType: "date",
          description: "must be a date and is required"
        }
      }
    }
  }
})


// 1. Geospatial index to power $near and $geoWithin queries
db.stops.createIndex({ location: "2dsphere" })

// 2. Compound index covers city-only lookups (prefix rule) AND the common city+RTC query
db.stops.createIndex({ city: 1, rtc: 1 })

// 3. Separate RTC-only index for querying all stops for an RTC regardless of city
db.stops.createIndex({ rtc: 1 })

db.createCollection("buses", {
  validator: {
    $jsonSchema: {
      bsonType: "object",
      required: ["_id", "routeId", "rtc", "routeName", "registrationNumber", "capacity", "isActive", "createdAt", "updatedAt"],
      properties: {
        _id: {
          bsonType: "string",
          description: "must be a string (natural key) and is required"
        },
        routeId: {
          bsonType: "objectId",
          description: "must be an ObjectId referencing Routes and is required"
        },
        rtc: {
          enum: ["GSRTC", "MSRTC", "RSRTC"],
          description: "must be one of the specified RTCs and is required"
        },
        routeName: {
          bsonType: "string",
          description: "must be a string and is required"
        },
        registrationNumber: {
          bsonType: "string",
          description: "must be a string (official reg) and is required"
        },
        capacity: {
          bsonType: ["int", "double", "long"],
          description: "must be a number representing seating capacity"
        },
        driverId: {
          bsonType: ["objectId", "null"],
          description: "must be an ObjectId referencing Drivers or null"
        },
        isActive: {
          bsonType: "bool",
          description: "must be a boolean"
        },
        // GeoJSON Point — enables $near queries (e.g. "find buses within 5km of stop X")
        // NOTE: required sub-fields are NOT listed here — the field itself can be null (bus off-shift),
        // and MongoDB $jsonSchema cannot enforce sub-field required when the parent is nullable.
        lastKnownLocation: {
          bsonType: ["object", "null"],
          properties: {
            type: {
              enum: ["Point"],
              description: "must be the string 'Point'"
            },
            coordinates: {
              bsonType: "array",
              minItems: 2,
              maxItems: 2,
              items: { bsonType: ["double", "int", "long", "decimal"] },
              description: "must be [longitude, latitude] — GeoJSON order"
            },
            speed_kmh: { bsonType: ["double", "int", "decimal"] },
            heading_deg: { bsonType: ["double", "int", "decimal"] },
            recordedAt: { bsonType: "date" }
          }
        },
        createdAt: { bsonType: "date" },
        updatedAt: { bsonType: "date" }
      }
    }
  }
})

// 1. Compound index to quickly filter active buses for a specific RTC
db.buses.createIndex({ rtc: 1, isActive: 1 })

// 2. Standard index to find all buses assigned to a specific route
db.buses.createIndex({ routeId: 1 })

// 3. Standard index to see which bus a driver is assigned to
db.buses.createIndex({ driverId: 1 })

// 4. Geospatial index — powers $near queries ("find all active buses within 5km of stop X")
//    sparse: true — skips documents where lastKnownLocation is absent (buses not yet on shift)
db.buses.createIndex({ lastKnownLocation: "2dsphere" }, { sparse: true })

// 5. Index on the nested recordedAt for telemetry sorting/cleanup
db.buses.createIndex({ "lastKnownLocation.recordedAt": 1 })

// 6. Unique index to prevent duplicate vehicle registrations
db.buses.createIndex({ registrationNumber: 1 }, { unique: true })


db.createCollection("routes", {
  validator: {
    $jsonSchema: {
      bsonType: "object",
      required: ["name", "rtc", "stopIds", "totalDistanceKm", "estimatedDurationMin", "isActive", "createdAt", "updatedAt"],
      properties: {
        name: {
          bsonType: "string",
          description: "must be a string and is required"
        },
        rtc: {
          enum: ["GSRTC", "MSRTC", "RSRTC"],
          description: "must be one of the specified RTCs and is required"
        },
        stopIds: {
          bsonType: "array",
          description: "must be an array of ObjectIds representing the ordered stops",
          items: {
            bsonType: "objectId"
          }
        },
        totalDistanceKm: {
          bsonType: ["double", "int", "decimal"],
          description: "must be a number representing the total distance"
        },
        estimatedDurationMin: {
          bsonType: ["double", "int", "decimal"],
          description: "must be a number representing estimated duration"
        },
        isActive: {
          bsonType: "bool",
          description: "must be a boolean"
        },
        createdAt: {
          bsonType: "date"
        },
        updatedAt: {
          bsonType: "date"
        }
      }
    }
  }
})

// 1. Unique compound index: a route name must be unique within its RTC
//    (GSRTC and MSRTC can both have "Ahmedabad-Surat", but GSRTC cannot have it twice)
db.routes.createIndex({ rtc: 1, name: 1 }, { unique: true })

// 2. Multikey index: MongoDB indexes every individual ObjectId inside the array
db.routes.createIndex({ stopIds: 1 })

// 3. Compound index to quickly filter active routes for a given RTC
db.routes.createIndex({ rtc: 1, isActive: 1 })

db.createCollection("shifts", {
  validator: {
    $jsonSchema: {
      bsonType: "object",
      required: ["driverId", "busId", "startedAt", "startLocation"],
      properties: {
        driverId: {
          bsonType: "objectId",
          description: "must be an ObjectId referencing the driver and is required"
        },
        busId: {
          bsonType: "string",
          description: "must be a string referencing the Bus natural key and is required"
        },
        startedAt: {
          bsonType: "date",
          description: "must be a date and is required"
        },
        endedAt: {
          bsonType: ["date", "null"],
          description: "must be a date or null if the shift is active"
        },
        durationMin: {
          bsonType: ["double", "int", "decimal", "null"],
          description: "must be a number computed on shift end"
        },
        totalPointsRecorded: {
          bsonType: ["int", "long", "null"],
          description: "must be a number tracking GPS updates"
        },
        startLocation: {
          bsonType: "object",
          required: ["lat", "lng"],
          properties: {
            lat: { bsonType: ["double", "int", "decimal"] },
            lng: { bsonType: ["double", "int", "decimal"] }
          }
        },
        endLocation: {
          bsonType: ["object", "null"],
          properties: {
            lat: { bsonType: ["double", "int", "decimal"] },
            lng: { bsonType: ["double", "int", "decimal"] }
          }
        }
      }
    }
  }
})

// 1. Compound index: Find a specific driver's shifts, sorted newest to oldest
db.shifts.createIndex({ driverId: 1, startedAt: -1 })

// 2. Compound index: Find a specific bus's shift history, sorted newest to oldest
db.shifts.createIndex({ busId: 1, startedAt: -1 })

// 3. Standard index: Quickly query for all currently active shifts ({ endedAt: null })
db.shifts.createIndex({ endedAt: 1 })
